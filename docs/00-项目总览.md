const p = dv.current();
const today = moment().startOf("day");

function fmtDate(value) {
  if (!value) return "—";
  const date = moment(String(value)).startOf("day");
  return date.isValid() ? date.format("YYYY-MM-DD") : String(value);
}

function timelineProgress(startValue, endValue) {
  if (!startValue || !endValue) return { pct: null, label: "未设置" };
  const start = moment(String(startValue)).startOf("day");
  const end = moment(String(endValue)).startOf("day");
  if (!start.isValid() || !end.isValid() || !end.isAfter(start)) {
    return { pct: null, label: "日期无效" };
  }
  if (today.isSameOrBefore(start)) return { pct: 0, label: "0%" };
  if (today.isSameOrAfter(end)) return { pct: 100, label: "100%" };
  const totalDays = end.diff(start, "days");
  const elapsedDays = today.diff(start, "days");
  const pct = Math.max(0, Math.min(100, Math.round((elapsedDays / totalDays) * 100)));
  return { pct, label: `${pct}%` };
}

function bar(pct) {
  if (pct === null || Number.isNaN(Number(pct))) return "—";
  const filled = Math.max(0, Math.min(10, Math.round(Number(pct) / 10)));
  return `${"●".repeat(filled)}${"○".repeat(10 - filled)} ${pct}%`;
}

const actualPct = Number(p.progress_pct ?? 0);
const actualLabel = p.progress_label ?? `${actualPct}%`;
const timeline = timelineProgress(p.start_date, p.end_date);

dv.table(
